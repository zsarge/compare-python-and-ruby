====RUBY=====
loop do
  a = rand(20)
  print a
  if a == 10
    puts
    break
  end
  b = rand(20)
  puts "\t#{b}"
end

loop do
  print a = rand(20)
  puts or break if a == 10
  puts "\t#{rand(20)}"
end
